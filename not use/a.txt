
import sys
import os
import time

sys.path.append(os.path.abspath(os.path.join(os.path.dirname(__file__), "..")))

import streamlit as st
import cv2
import numpy as np
import webbrowser
from tensorflow.keras.models import load_model
from llm.text_emotion import detect_emotion_from_text

# ===================== CONFIG =====================
st.set_page_config(
    page_title="Emotion-Based Music Recommendation",
    layout="centered"
)

# ===================== LOAD MODEL =====================
MODEL_PATH = "../cnn/emotion_model.h5"
model = load_model(MODEL_PATH)

emotion_labels = [
    "Angry", "Disgust", "Fear",
    "Happy", "Sad", "Surprise", "Neutral"
]

# ===================== SESSION STATE =====================
if "detected_emotion" not in st.session_state:
    st.session_state.detected_emotion = None

if "stable_emotion" not in st.session_state:
    st.session_state.stable_emotion = None

if "emotion_start_time" not in st.session_state:
    st.session_state.emotion_start_time = None

if "song_recommended" not in st.session_state:
    st.session_state.song_recommended = False

# ===================== FUNCTIONS =====================
def recommend_song_on_youtube(emotion):
    query = f"{emotion} mood songs playlist"
    url = f"https://www.youtube.com/results?search_query={query.replace(' ', '+')}"
    webbrowser.open(url)

def predict_emotion_from_face(face_img):
    face_img = cv2.resize(face_img, (48, 48))
    face_img = face_img / 255.0
    face_img = np.reshape(face_img, (1, 48, 48, 1))

    prediction = model.predict(face_img, verbose=0)

    confidence = np.max(prediction)
    emotion = emotion_labels[np.argmax(prediction)]

    if confidence < 0.5:
        emotion = "Uncertain"

    return emotion

# ===================== UI =====================
st.title("Emotion-Based Music Recommendation System")
st.write("Choose how you want to express your emotion:")

mode = st.radio("Select Input Method", ["Use Camera", "Type Text"])

# ===================== CAMERA MODE =====================
if mode == "Use Camera":
    st.subheader("📷 Live Camera Emotion Detection")

    start_camera = st.checkbox("Start Camera")

    frame_window = st.image([])
    emotion_display = st.empty()

    if start_camera:
        cap = cv2.VideoCapture(0)
        face_cascade = cv2.CascadeClassifier(
            cv2.data.haarcascades + "haarcascade_frontalface_default.xml"
        )

        while start_camera:
            ret, frame = cap.read()
            if not ret:
                st.warning("Camera not accessible")
                break

            gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
            faces = face_cascade.detectMultiScale(gray, 1.3, 5)

            for (x, y, w, h) in faces:
                face_roi = gray[y:y+h, x:x+w]
                emotion = predict_emotion_from_face(face_roi)

                current_time = time.time()

                # If emotion changes → reset timer
                if st.session_state.stable_emotion != emotion:
                    st.session_state.stable_emotion = emotion
                    st.session_state.emotion_start_time = current_time
                    st.session_state.song_recommended = False

                # Draw face box & emotion
                cv2.rectangle(frame, (x, y), (x+w, y+h), (0, 255, 0), 2)
                cv2.putText(
                    frame, emotion, (x, y-10),
                    cv2.FONT_HERSHEY_SIMPLEX, 0.9, (0, 255, 0), 2
                )

                #  Emotion stability check
                if (current_time - st.session_state.emotion_start_time) >= 3:
                    st.session_state.detected_emotion = emotion

                    if not st.session_state.song_recommended:
                        st.success(f"Final Emotion Detected: {emotion}")
                        recommend_song_on_youtube(emotion)
                        st.session_state.song_recommended = True

                    # STOP CAMERA AFTER ONE DETECTION
                    cap.release()
                    break

            frame_window.image(frame, channels="BGR")

            if st.session_state.detected_emotion:
                emotion_display.markdown(
                    f"### Detected Emotion: **{st.session_state.detected_emotion}**"
                )

        cap.release()

# ===================== TEXT MODE =====================
if mode == "Type Text":
    st.subheader("📝 Text-Based Emotion Detection")

    user_text = st.text_area(
        "Type how you are feeling:",
        placeholder="Example: I feel calm and relaxed today"
    )

    if st.button("Analyze Emotion"):
        if user_text.strip():
            emotion = detect_emotion_from_text(user_text)

            st.session_state.detected_emotion = emotion
            st.session_state.song_recommended = False

            st.success(f"Detected Emotion: **{emotion}**")

            # ⏳ Small delay for UX
            time.sleep(2)
            recommend_song_on_youtube(emotion)
            st.session_state.song_recommended = True
        else:
            st.warning("Please enter some text.")

# ===================== RESET =====================
st.markdown("---")
if st.button("🔄 Reset"):
    st.session_state.detected_emotion = None
    st.session_state.stable_emotion = None
    st.session_state.emotion_start_time = None
    st.session_state.song_recommended = False
    st.success("System reset successfully")

# ===================== QUIT =====================
st.markdown("---")
if st.button("❌ Quit Application"):
    st.stop()
